  ```typescript
  import { eq, isNull, asc } from 'drizzle-orm';
  import { getDb } from '../client';
  import { syncQueue } from '../schema';
  import { generateUUID } from '../../utils/uuid';

  export const syncQueueRepo = {
    /** Get all pending (unsynced) items ordered by createdAt ASC for causal ordering */
    async getPending(): Promise<SyncQueueItem[]> {
      const db = getDb();
      return db.select().from(syncQueue)
        .where(isNull(syncQueue.syncedAt))
        .orderBy(asc(syncQueue.createdAt));
    },

    /** Enqueue a new sync operation */
    async enqueue(entityType: string, entityId: string, operation: string, payload: object): Promise<void> {
      const db = getDb();
      await db.insert(syncQueue).values({
        id: generateUUID(),
        entityType,
        entityId,
        operation,
        payload: JSON.stringify(payload),
        createdAt: new Date().toISOString(),
      });
    },

    /** Mark a queue item as successfully synced */
    async markSynced(id: string): Promise<void> {
      const db = getDb();
      await db.update(syncQueue)
        .set({ syncedAt: new Date().toISOString() })
        .where(eq(syncQueue.id, id));
    },

    /** Record a failed sync attempt — increment retryCount, set lastError */
    async markFailed(id: string, errorMessage: string): Promise<void> {
      const db = getDb();
      const items = await db.select().from(syncQueue).where(eq(syncQueue.id, id));
      const currentRetry = items[0]?.retryCount ?? 0;
      await db.update(syncQueue)
        .set({ retryCount: currentRetry + 1, lastError: errorMessage })
        .where(eq(syncQueue.id, id));
    },

    /** Delete all synced items (cleanup) */
    async purgeSynced(): Promise<void> {
      const db = getDb();
      const { sql } = await import('drizzle-orm');
      await db.delete(syncQueue).where(sql`${syncQueue.syncedAt} IS NOT NULL`);
    },

    /** Clear all queue items (used during account deletion) */
    async clearAll(): Promise<void> {
      const db = getDb();
      await db.delete(syncQueue);
    },
  };
  ```

     ```typescript
     import NetInfo from '@react-native-community/netinfo';
     import { supabase } from '../lib/supabase';
     import { useAuthStore } from '../stores/authStore';
     import { syncQueueRepo } from '../db/repos/syncQueueRepo';
     import { assertSyncable } from './privacy-gate';

     const MAX_RETRIES = 5;

     export async function flushQueue(): Promise<void> {
       const { isAuthenticated, userId } = useAuthStore.getState();
       if (!isAuthenticated) return;

       const netState = await NetInfo.fetch();
       if (!netState.isConnected) return;

       useAuthStore.getState().setSyncStatus('syncing');

       try {
         const pending = await syncQueueRepo.getPending();
         let hasError = false;

         for (const item of pending) {
           if (item.retryCount >= MAX_RETRIES) continue; // skip permanently failed

           try {
             assertSyncable(item.entityType);
             const payload = JSON.parse(item.payload);

             if (item.operation === 'DELETE') {
               const { error } = await supabase
                 .from(item.entityType)
                 .delete()
                 .eq('id', item.entityId);
               if (error) throw error;
             } else {
               const { error } = await supabase
                 .from(item.entityType)
                 .upsert(payload, { onConflict: 'id' });
               if (error) throw error;
             }

             await syncQueueRepo.markSynced(item.id);
           } catch (err: any) {
             hasError = true;
             await syncQueueRepo.markFailed(item.id, err?.message ?? 'Unknown error');
           }
         }

         if (hasError) {
           useAuthStore.getState().setSyncStatus('error');
         } else {
           useAuthStore.getState().setSyncStatus('synced');
           useAuthStore.getState().setLastSyncedAt(new Date().toISOString());
         }
       } catch {
         useAuthStore.getState().setSyncStatus('error');
       }
     }

     /** Convenience wrapper for manual "Sync Now" button */
     export async function syncNow(): Promise<void> {
       return flushQueue();
     }

     /** Clean up already-synced queue entries */
     export async function purgeCompleted(): Promise<void> {
       await syncQueueRepo.purgeSynced();
     }
     ```

    Export functions with these EXACT formulas:

    **canStartBattle(currentLevel, activeBattle, lastBattleEndedAt, now, cooldownDays):**
    - if currentLevel < 10 return false
    - if activeBattle?.status === 'active' return false
    - if !lastBattleEndedAt return true
    - return (Date(now) - Date(lastBattleEndedAt)) >= cooldownDays * 86400000

    **calculateBossMaxHp(playerLevel):**
    - return 100 + (playerLevel - 10) * 15
    - Level 10 = 100, Level 20 = 250, Level 30 = 400

    **calculateBossXpReward(playerLevel):**
    - return Math.min(Math.max(200 + (playerLevel - 10) * 15, 200), 500)

    **getMaxDays(playerLevel):**
    - Base 7 days. At level 20+, reduce to 6. At level 30+, reduce to 5.
    - return playerLevel >= 30 ? 5 : playerLevel >= 20 ? 6 : 7

    **calculateDailyDamage(habitsCompleted, totalHabits, bossMaxHp):**
    - if totalHabits === 0 return 0
    - return Math.round(bossMaxHp * 0.20 * (habitsCompleted / totalHabits))

    **calculateDailyHealing(habitsMissed, totalHabits, bossMaxHp, mercyModeActive):**
    - if totalHabits === 0 return 0
    - let healing = Math.round(bossMaxHp * 0.10 * (habitsMissed / totalHabits))
    - if (mercyModeActive) healing = Math.round(healing * 0.5)
    - return healing

    **applyDailyOutcome(currentHp, bossMaxHp, damage, healing):**
    - return Math.min(Math.max(currentHp - damage + healing, 0), bossMaxHp)

    **calculatePartialXp(fullXpReward, bossMaxHp, bossHp):**
    - const damageDealt = bossMaxHp - bossHp
    - return Math.round(fullXpReward * (damageDealt / bossMaxHp))

    **getBossDialoguePhase(isFirstDay, bossHpRatio, battleStatus):**
    - if battleStatus === 'defeated' return 'defeated'
    - if isFirstDay return 'intro'
    - if bossHpRatio < 0.5 return 'player_winning'
    - return 'taunt'

    **suggestArchetype(habitCompletionRates, recentMissPatterns):**
    - If no data (empty objects/arrays), return 'procrastinator' as safe default
    - Simple heuristic: map most-missed habit categories to archetypes
    - Return ArchetypeId

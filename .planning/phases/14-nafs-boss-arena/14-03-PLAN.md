    **Create src/stores/bossStore.ts:**

      canStartBattle, calculateBossMaxHp, calculateBossXpReward,

    Interface:
    ```typescript
    interface BossState {
      activeBattle: BossBattle | null;
      loading: boolean;
      pendingDefeatCelebration: { archetype: ArchetypeId; xpAwarded: number } | null;
      pendingEscapeNotice: { archetype: ArchetypeId; damagePercent: number; xpAwarded: number } | null;

      loadActiveBattle: (userId: string) => Promise<void>;
      startBattle: (archetype: ArchetypeId, playerLevel: number, userId: string) => Promise<boolean>;
      processDailyOutcome: (userId: string, habitsCompleted: number, totalHabits: number) => Promise<void>;
      abandonBattle: (userId: string) => Promise<void>;
      canStart: (userId: string) => Promise<boolean>;
      clearCelebration: () => void;
      clearEscapeNotice: () => void;
    }
    ```

    **loadActiveBattle(userId):**
    1. set({ loading: true })
    2. const battle = await bossRepo.getActiveBattle(userId)
    3. If no battle, set({ activeBattle: null, loading: false }), return
    4. **Multi-day catch-up:** Compare battle.updatedAt date to today (YYYY-MM-DD)
       - daysDiff = number of calendar days between updatedAt and now
       - For each missed day: treat as 0 habits completed (app was closed)
       - Calculate healing per missed day using calculateDailyHealing(totalHabits=battle.bossMaxHp field context, habitsMissed=all, mercyMode=battle.mercyMode)
       - Actually: for days the app was closed, we don't know totalHabits. Use a conservative fixed miss: healing = calculateDailyHealing(1, 1, battle.bossMaxHp, battle.mercyMode) — full miss for each untracked day
       - Apply each day's outcome, append to dailyLog, increment currentDay
       - If bossHp <= 0: trigger defeat flow
       - If currentDay > maxDays: trigger escape flow
       - Write all updates via bossRepo.updateDailyOutcome
    5. Refresh: re-read from DB, set({ activeBattle: freshBattle, loading: false })

    **startBattle(archetype, playerLevel, userId):**
    1. Check canStartBattle via engine function
    2. const bossMaxHp = calculateBossMaxHp(playerLevel)
    3. const maxDays = getMaxDays(playerLevel)
    4. const mercyMode = useSettingsStore.getState().mercyModeActive
    5. const id = Crypto.randomUUID()
    6. const now = new Date().toISOString()
    7. await bossRepo.create({ id, userId, archetype, bossHp: bossMaxHp, bossMaxHp, currentDay: 1, maxDays, dailyLog: '[]', status: 'active', mercyMode, startedAt: now, updatedAt: now })
    8. Reload active battle
    9. Return true (or false if eligibility check failed)

    **processDailyOutcome(userId, habitsCompleted, totalHabits):**
    1. const battle = get().activeBattle; if !battle return
    2. Check if today's date already processed (compare battle.updatedAt YYYY-MM-DD to today)
    3. If already processed today, return (guard against double-processing)
    4. const damage = calculateDailyDamage(habitsCompleted, totalHabits, battle.bossMaxHp)
    5. const missed = totalHabits - habitsCompleted
    6. const healing = calculateDailyHealing(missed, totalHabits, battle.bossMaxHp, battle.mercyMode)
    7. const newHp = applyDailyOutcome(battle.bossHp, battle.bossMaxHp, damage, healing)
    8. const newDay = battle.currentDay + 1
    9. Parse dailyLog JSON, append new DailyLogEntry, stringify
    10. await bossRepo.updateDailyOutcome(battle.id, newHp, newDay, newDailyLog, new Date().toISOString())
    11. If newHp <= 0: **defeat flow**
        - await bossRepo.defeat(battle.id, now)
        - const fullXp = calculateBossXpReward(playerLevel from gameStore)
        - await useGameStore.getState().awardXP(userId, fullXp, 1.0, 'boss_defeat', battle.id)
        - await useGameStore.getState().checkTitles(userId)
        - set({ activeBattle: null, pendingDefeatCelebration: { archetype: battle.archetype as ArchetypeId, xpAwarded: fullXp } })
    12. Else if newDay > battle.maxDays: **escape flow**
        - const partialXp = calculatePartialXp(calculateBossXpReward(playerLevel), battle.bossMaxHp, newHp)
        - const damagePercent = Math.round(((battle.bossMaxHp - newHp) / battle.bossMaxHp) * 100)
        - await bossRepo.escape(battle.id, now)
        - await useGameStore.getState().awardXP(userId, partialXp, 1.0, 'boss_defeat', battle.id)
        - set({ activeBattle: null, pendingEscapeNotice: { archetype: battle.archetype as ArchetypeId, damagePercent, xpAwarded: partialXp } })
    13. Else: reload active battle from DB

    **abandonBattle(userId):**
    1. Same as escape flow but player-initiated
    2. Uses current battle state for partial XP calculation

    **canStart(userId):**
    1. const lastBattle = await bossRepo.getLastBattle(userId)
    2. const activeBattle = await bossRepo.getActiveBattle(userId)
    3. const level = useGameStore.getState().currentLevel
    4. return canStartBattle(level, activeBattle, lastBattle?.endedAt ?? null, new Date().toISOString(), 3)

    **clearCelebration():** set({ pendingDefeatCelebration: null })
    **clearEscapeNotice():** set({ pendingEscapeNotice: null })
